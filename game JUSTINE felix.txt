    (function(){
        // CANVAS
        const canvas = document.getElementById('gameCanvas');

        // ==================== GAME STATE ====================

        let health = 20;
        let maxHealth = 20;   // untuk UI bar
        let enemies = [];
        let bullets = [];
        let bosses = [];        
        let enemyId = 0;
        let frameCount = 0;
        let lastSpawnFrame = 0;
        let gameWin = false;
        let finalBossSpawned = false;   
        
        // Parameter mode dinamis
        let playerHitRadius = 26;
        let enemySpeedBase = 1.5;
        let enemySpeedRandomExtra = 1.8;
        let spawnBaseDelay = 32;
        let maxEnemiesAtOnce = 12;
        let extraFastOnScore = true;
        
        // Background objects
        let stars = [];
        let planets = [];
        let nebulae = [];
        
        // AIM

        // UI Elements
        const scoreSpan = document.getElementById('scoreValue');
        const healthSpan = document.getElementById('healthValue');

        const ENEMY_COLORS = ["#e34234", "#e25822", "#db3d2f", "#c73b2b", "#ff6b4a"];
        const BOSS_COLORS = ["#9b2f1f", "#b13e2a", "#7a2417"];
        
        // ==================== UPDATE UI DARAH ====================
        function updateHealthUI() {
            healthSpan.innerText = health;
            let percent = (health / maxHealth) * 100;
            percent = Math.max(0, percent);
            healthBarFill.style.width = percent + "%";
            // change color based on health
            if (percent < 25) {
                healthBarFill.style.background = "linear-gradient(90deg, #ff0000, #ff6666)";
            } else if (percent < 60) {
                healthBarFill.style.background = "linear-gradient(90deg, #ff6600, #ffaa66)";
            } else {
                healthBarFill.style.background = "linear-gradient(90deg, #ff4d4d, #ff9999)";
            }
        }
        
        // ==================== BACKGROUND DETAIL ====================
        function initBackground() {
            for(let i=0; i<450; i++) {
                stars.push({
                    x: Math.random() * width,
                    y: Math.random() * height,
                    radius: Math.random() * 2.5 + 0.5,
                    alpha: Math.random() * 0.7 + 0.3,
                    twinkleSpeed: 0.02 + Math.random() * 0.03,
                    phase: Math.random() * Math.PI * 2
                });
            }
            planets.push({ x: width*0.12, y: height*0.18, radius: 48, color: "#8B5A2B", ring: true });
            planets.push({ x: width*0.88, y: height*0.75, radius: 38, color: "#4A7A9C", ring: false });
            planets.push({ x: width*0.73, y: height*0.12, radius: 28, color: "#B85C38", ring: false });
            planets.push({ x: width*0.06, y: height*0.88, radius: 32, color: "#5D6B7A", ring: true });
            planets.push({ x: width*0.5, y: height*0.92, radius: 22, color: "#9B6A4A", ring: false });
            
            for(let i=0; i<6; i++) {
                nebulae.push({
                    x: Math.random() * width,
                    y: Math.random() * height,
                    radius: 90 + Math.random() * 110,
                    color: `hsla(${Math.random() * 60 + 200}, 70%, 55%, 0.07)`
                });
            }
        }
        
        function drawBackground() {
            const grad = ctx.createLinearGradient(0, 0, width, height);
            grad.addColorStop(0, "#03071e");
            grad.addColorStop(1, "#0a0f2a");
            ctx.fillStyle = grad;
            ctx.fillRect(0, 0, width, height);
            
            for(let n of nebulae) {
                ctx.beginPath();
                ctx.arc(n.x, n.y, n.radius, 0, Math.PI*2);
                ctx.fillStyle = n.color;
                ctx.fill();
            }
            
            for(let p of planets) {
                ctx.shadowBlur = 15;
                ctx.shadowColor = "rgba(0,0,0,0.5)";
                ctx.beginPath();
                ctx.arc(p.x, p.y, p.radius, 0, Math.PI*2);
                ctx.fillStyle = p.color;
                ctx.fill();
                ctx.shadowBlur = 0;
                ctx.beginPath();
                ctx.arc(p.x - p.radius*0.2, p.y - p.radius*0.1, p.radius*0.15, 0, Math.PI*2);
                ctx.fillStyle = "rgba(0,0,0,0.3)";
                ctx.fill();
                ctx.beginPath();
                ctx.arc(p.x + p.radius*0.25, p.y + p.radius*0.15, p.radius*0.12, 0, Math.PI*2);
                ctx.fill();
                if(p.ring) {
                    ctx.beginPath();
                    ctx.ellipse(p.x, p.y, p.radius*1.4, p.radius*0.4, 0.3, 0, Math.PI*2);
                    ctx.fillStyle = "rgba(218,165,32,0.4)";
                    ctx.fill();
                    ctx.beginPath();
                    ctx.ellipse(p.x, p.y, p.radius*1.5, p.radius*0.35, 0.3, 0, Math.PI*2);
                    ctx.fillStyle = "rgba(255,215,0,0.2)";
                    ctx.fill();
                }
                ctx.beginPath();
                ctx.arc(p.x - p.radius*0.3, p.y - p.radius*0.3, p.radius*0.2, 0, Math.PI*2);
                ctx.fillStyle = "rgba(255,255,255,0.15)";
                ctx.fill();
            }
            
            for(let s of stars) {
                let twinkle = Math.sin(frameCount * s.twinkleSpeed + s.phase) * 0.3 + 0.7;
                ctx.beginPath();
                ctx.arc(s.x, s.y, s.radius, 0, Math.PI*2);
                ctx.fillStyle = `rgba(255, 240, 200, ${s.alpha * twinkle})`;
                ctx.fill();
            }
            ctx.shadowBlur = 0;
        }
        
        // ==================== ANIMASI MUSUH DETAIL ====================
        function drawEnemy(e, isBoss = false) {
            ctx.save();
            ctx.shadowBlur = isBoss ? 15 : 8;
            ctx.shadowColor = isBoss ? "#ff6600" : "#ff4400";
            ctx.beginPath();
            ctx.arc(e.x, e.y, e.radius, 0, Math.PI*2);
            const grad = ctx.createRadialGradient(e.x-5, e.y-5, 3, e.x, e.y, e.radius);
            grad.addColorStop(0, e.color);
            grad.addColorStop(1, isBoss ? "#4a0101" : "#621a0c");
            ctx.fillStyle = grad;
            ctx.fill();
            
            const wingAngle = Math.sin(frameCount * 0.1 + e.id) * 0.3;
            ctx.beginPath();
            ctx.moveTo(e.x - e.radius*0.7, e.y - e.radius*0.3);
            ctx.lineTo(e.x - e.radius*1.1, e.y - e.radius*0.5 + wingAngle*5);
            ctx.lineTo(e.x - e.radius*0.6, e.y + e.radius*0.1);
            ctx.fillStyle = e.color;
            ctx.fill();
            ctx.beginPath();
            ctx.moveTo(e.x + e.radius*0.7, e.y - e.radius*0.3);
            ctx.lineTo(e.x + e.radius*1.1, e.y - e.radius*0.5 + wingAngle*5);
            ctx.lineTo(e.x + e.radius*0.6, e.y + e.radius*0.1);
            ctx.fill();
            
            const blink = Math.sin(frameCount * 0.15) > 0.9 ? 0 : 1;
            ctx.fillStyle = "#ffffff";
            ctx.beginPath();
            ctx.arc(e.x - e.radius*0.35, e.y - e.radius*0.2, e.radius*0.22, 0, Math.PI*2);
            ctx.fill();
            ctx.beginPath();
            ctx.arc(e.x + e.radius*0.35, e.y - e.radius*0.2, e.radius*0.22, 0, Math.PI*2);
            ctx.fill();
            ctx.fillStyle = "#000000";
            if(blink) {
                ctx.beginPath();
                ctx.arc(e.x - e.radius*0.4, e.y - e.radius*0.25, e.radius*0.12, 0, Math.PI*2);
                ctx.fill();
                ctx.beginPath();
                ctx.arc(e.x + e.radius*0.3, e.y - e.radius*0.25, e.radius*0.12, 0, Math.PI*2);
                ctx.fill();
            } else {
                ctx.fillRect(e.x - e.radius*0.48, e.y - e.radius*0.32, e.radius*0.18, e.radius*0.12);
                ctx.fillRect(e.x + e.radius*0.3, e.y - e.radius*0.32, e.radius*0.18, e.radius*0.12);
            }
            
            ctx.beginPath();
            ctx.ellipse(e.x, e.y + e.radius*0.25, e.radius*0.25, e.radius*0.12, 0, 0, Math.PI*2);
            ctx.fillStyle = "#3a180e";
            ctx.fill();
            ctx.beginPath();
            ctx.ellipse(e.x, e.y + e.radius*0.23, e.radius*0.12, e.radius*0.06, 0, 0, Math.PI*2);
            ctx.fillStyle = "#ff6666";
            ctx.fill();
            
            const hornAngle = Math.sin(frameCount * 0.12) * 3;
            ctx.beginPath();
            ctx.moveTo(e.x - e.radius*0.5, e.y - e.radius*0.7);
            ctx.lineTo(e.x - e.radius*0.7 + hornAngle, e.y - e.radius*1.1);
            ctx.lineTo(e.x - e.radius*0.3, e.y - e.radius*0.65);
            ctx.fillStyle = "#8B4513";
            ctx.fill();
            ctx.beginPath();
            ctx.moveTo(e.x + e.radius*0.5, e.y - e.radius*0.7);
            ctx.lineTo(e.x + e.radius*0.7 - hornAngle, e.y - e.radius*1.1);
            ctx.lineTo(e.x + e.radius*0.3, e.y - e.radius*0.65);
            ctx.fill();
            
            if(isBoss) {
                ctx.beginPath();
                ctx.arc(e.x, e.y, e.radius + 5, 0, Math.PI*2);
                ctx.fillStyle = "rgba(255,50,0,0.2)";
                ctx.fill();
                for(let i=0; i<5; i++) {
                    let angle = (frameCount * 0.05 + i * 1.2);
                    let xOff = Math.cos(angle) * (e.radius + 8);
                    let yOff = Math.sin(angle) * 8 - 10;
                    ctx.beginPath();
                    ctx.arc(e.x + xOff, e.y + yOff, 5, 0, Math.PI*2);
                    ctx.fillStyle = "#ffcc44";
                    ctx.fill();
                }
                const hpPercent = e.hp / e.maxHp;
                ctx.fillStyle = "#330000";
                ctx.fillRect(e.x - e.radius, e.y - e.radius - 15, e.radius*2, 10);
                ctx.fillStyle = "#ff4444";
                ctx.fillRect(e.x - e.radius, e.y - e.radius - 15, e.radius*2 * hpPercent, 10);
                ctx.fillStyle = "white";
                ctx.font = "bold 12px monospace";
                ctx.fillText(`${e.hp}/${e.maxHp}`, e.x - 18, e.y - e.radius - 18);
                if(e.isFinal) {
                    ctx.font = "bold 14px monospace";
                    ctx.fillStyle = "#ffaa66";
                    ctx.fillText("☠️ ONE SHOT KILL ☠️", e.x - 55, e.y - e.radius - 32);
                }
            }
            ctx.restore();
        }
        
        // ==================== EFEK TEMBAKAN ====================
        function addShootEffect(x, y, isHit = false) {
            let count = isHit ? 14 : 7;
            for(let i=0; i<count; i++) {
                bullets.push({
                    x: x, y: y,
                    radius: isHit ? (Math.random() * 12 + 4) : (Math.random() * 7 + 2),
                    life: 0.85,
                    vx: (Math.random() - 0.5) * (isHit ? 4.5 : 2),
                    vy: (Math.random() - 0.5) * (isHit ? 4.5 : 2),
                    color: `hsl(${Math.random()*50 + 20}, 100%, 60%)`
                });
            }
        }
        
        // ==================== GAME MECHANICS ====================
        function applyDifficultySettings() {
            if (currentDifficulty === 'easy') {
                maxHealth = 20; health = 20; playerHitRadius = 32; enemySpeedBase = 1.0; enemySpeedRandomExtra = 1.2;
                spawnBaseDelay = 45; maxEnemiesAtOnce = 9; extraFastOnScore = false;
            } 
            else if (currentDifficulty === 'normal') {
                maxHealth = 10; health = 10; playerHitRadius = 26; enemySpeedBase = 1.4; enemySpeedRandomExtra = 1.6;
                spawnBaseDelay = 32; maxEnemiesAtOnce = 12; extraFastOnScore = true;
            } 
            else if (currentDifficulty === 'hard') {
                maxHealth = 1; health = 1; playerHitRadius = 22; enemySpeedBase = 2.1; enemySpeedRandomExtra = 2.3;
                spawnBaseDelay = 22; maxEnemiesAtOnce = 16; extraFastOnScore = true;
            }
            updateHealthUI();
        }
        
        function applyDamage(amount, isFinalBossHit = false) {
            if (!gameRunning || gameWin) return;
            if (isFinalBossHit) { 
                health = 0; 
                updateHealthUI(); 
                gameRunning = false; 
                gameWin = false; 
                return; 
            }
            health -= amount;
            if (health <= 0) { 
                health = 0; 
                updateHealthUI(); 
                gameRunning = false; 
                gameWin = false; 
            } else {
                updateHealthUI();
            }
        }
        
        let lastBossScoreThreshold = -1;
        function trySpawnBoss(currentScore) {
            if (!gameRunning || gameWin || finalBossSpawned || currentScore >= 150) return false;
            if (currentScore >= 20 && currentScore % 20 === 0 && currentScore !== lastBossScoreThreshold) {
                if (bosses.length > 0) return false;
                lastBossScoreThreshold = currentScore;
                let side = Math.floor(Math.random() * 4);
                let x, y, radius = 44;
                if (side === 0) { x = Math.random() * (width - 100) + 50; y = -radius + 5; }
                else if (side === 1) { x = width + radius - 5; y = Math.random() * (height - 100) + 50; }
                else if (side === 2) { x = Math.random() * (width - 100) + 50; y = height + radius - 5; }
                else { x = -radius + 5; y = Math.random() * (height - 100) + 50; }
                const cx = width/2, cy = height/2;
                let dx = cx - x, dy = cy - y;
                const len = Math.hypot(dx, dy);
                let bossSpeed = (currentDifficulty === 'hard' ? 1.8 : (currentDifficulty === 'easy' ? 0.85 : 1.3));
                const vx = (dx / len) * bossSpeed;
                const vy = (dy / len) * bossSpeed;
                let bossHp = (currentDifficulty === 'easy' ? 6 : (currentDifficulty === 'hard' ? 14 : 8));
                if (currentScore >= 100) bossHp += 3;
                bosses.push({ id: enemyId++, x, y, radius, vx, vy, color: BOSS_COLORS[Math.floor(Math.random() * BOSS_COLORS.length)], hp: bossHp, maxHp: bossHp, isBoss: true, isFinal: false });
                return true;
            }
            return false;
        }
        
        function spawnFinalBoss() {
            if (finalBossSpawned || !gameRunning || gameWin) return false;
            if (score >= 150 && !finalBossSpawned) {
                enemies = [];
                bosses = bosses.filter(b => b.isFinal);
                finalBossSpawned = true;
                let radius = 60;
                let x = width/2 - 30, y = -radius + 10;
                const cx = width/2, cy = height/2;
                let dx = cx - x, dy = cy - y;
                const len = Math.hypot(dx, dy);
                const speed = 0.65;
                const vx = (dx / len) * speed;
                const vy = (dy / len) * speed;
                let finalBossHP = currentDifficulty === 'easy' ? 20 : (currentDifficulty === 'normal' ? 30 : 50);
                bosses.push({ id: enemyId++, x, y, radius, vx, vy, color: "#6a0d0d", hp: finalBossHP, maxHp: finalBossHP, isBoss: true, isFinal: true });
                return true;
            }
            return false;
        }
        
        function spawnEnemy() {
            if (!gameRunning || gameWin || finalBossSpawned) return;
            if (enemies.length + bosses.length > maxEnemiesAtOnce + 4) return;
            let radius = Math.floor(Math.random() * (24 - 12 + 1) + 12);
            let side = Math.floor(Math.random() * 4);
            let x, y;
            const pad = radius + 8;
            if (side === 0) { x = Math.random() * (width - pad*2) + pad; y = -radius + 2; }
            else if (side === 1) { x = width + radius - 2; y = Math.random() * (height - pad*2) + pad; }
            else if (side === 2) { x = Math.random() * (width - pad*2) + pad; y = height + radius - 2; }
            else { x = -radius + 2; y = Math.random() * (height - pad*2) + pad; }
            const cx = width/2, cy = height/2;
            let dx = cx - x, dy = cy - y;
            const len = Math.hypot(dx, dy);
            let speed = enemySpeedBase + Math.random() * enemySpeedRandomExtra;
            if (extraFastOnScore && currentDifficulty === 'hard' && score > 25) speed += 0.5;
            const vx = (dx / len) * speed;
            const vy = (dy / len) * speed;
            enemies.push({ id: enemyId++, x, y, radius, vx, vy, color: ENEMY_COLORS[Math.floor(Math.random()*ENEMY_COLORS.length)], hp: 1, isBoss: false });
        }
        
        function updateEntities() {
            for (let i=0; i<enemies.length; i++) {
                const e = enemies[i];
                e.x += e.vx; e.y += e.vy;
                const px = width/2, py = height/2;
                const dist = Math.hypot(e.x - px, e.y - py);
                if (dist < e.radius + playerHitRadius) {
                    if (gameRunning && !gameWin) { applyDamage(1, false); enemies.splice(i,1); i--; if (!gameRunning) break; }
                    else { enemies.splice(i,1); i--; }
                    continue;
                }
                if (e.x + e.radius < -350 || e.x - e.radius > width+350 || e.y + e.radius < -350 || e.y - e.radius > height+350) {
                    enemies.splice(i,1); i--;
                }
            }
            for (let i=0; i<bosses.length; i++) {
                const b = bosses[i];
                b.x += b.vx; b.y += b.vy;
                const px = width/2, py = height/2;
                const dist = Math.hypot(b.x - px, b.y - py);
                if (dist < b.radius + playerHitRadius) {
                    if (gameRunning && !gameWin) {
                        if (b.isFinal) applyDamage(0, true);
                        else applyDamage(2, false);
                        bosses.splice(i,1); i--;
                        if (!gameRunning) break;
                    } else { bosses.splice(i,1); i--; }
                    continue;
                }
                if (b.x + b.radius < -500 || b.x - b.radius > width+500 || b.y + b.radius < -500 || b.y - b.radius > height+500) {
                    bosses.splice(i,1); i--;
                }
            }
        }
        
        function shootAt(tx, ty) {
            if (!gameRunning || gameWin) return false;
            let finalBossIndex = -1;
            for (let i=0; i<bosses.length; i++) if (bosses[i].isFinal && Math.hypot(tx - bosses[i].x, ty - bosses[i].y) <= bosses[i].radius + 15) { finalBossIndex = i; break; }
            if (finalBossIndex !== -1) {
                const fb = bosses[finalBossIndex];
                fb.hp -= 1;
                addShootEffect(tx, ty, true);
                if (fb.hp <= 0) {
                    score += 50; scoreSpan.innerText = score;
                    addShootEffect(fb.x, fb.y, true);
                    bosses.splice(finalBossIndex,1);
                    gameWin = true; gameRunning = false;
                    if (navigator.vibrate) navigator.vibrate(200);
                } else { if (navigator.vibrate) navigator.vibrate(40); }
                return true;
            }
            let hitBossIndex = -1;
            for (let i=0; i<bosses.length; i++) if (Math.hypot(tx - bosses[i].x, ty - bosses[i].y) <= bosses[i].radius + 12) { hitBossIndex = i; break; }
            if (hitBossIndex !== -1) {
                const boss = bosses[hitBossIndex];
                boss.hp -= 1;
                addShootEffect(tx, ty, true);
                if (boss.hp <= 0) {
                    score += 15; scoreSpan.innerText = score;
                    addShootEffect(boss.x, boss.y, true);
                    bosses.splice(hitBossIndex,1);
                    if (score >= 150 && !finalBossSpawned) spawnFinalBoss();
                    if (navigator.vibrate) navigator.vibrate(80);
                } else { if (navigator.vibrate) navigator.vibrate(30); }
                return true;
            }
            let hitIndex = -1;
            let closest = 45;
            for (let i=0; i<enemies.length; i++) {
                const e = enemies[i];
                const dist = Math.hypot(tx - e.x, ty - e.y);
                if (dist <= e.radius + 8 && dist < closest) { closest = dist; hitIndex = i; }
            }
            if (hitIndex !== -1) {
                const enemy = enemies[hitIndex];
                addShootEffect(enemy.x, enemy.y, true);
                enemies.splice(hitIndex,1);
                score++; scoreSpan.innerText = score;
                if (score >= 150 && !finalBossSpawned) spawnFinalBoss();
                if (navigator.vibrate) navigator.vibrate(20);
                return true;
            } else {
                addShootEffect(tx, ty, false);
                return false;
            }
        }
        
        function updateBulletEffects() {
            for (let i=0; i<bullets.length; i++) {
                bullets[i].life -= 0.045;
                if (bullets[i].vx) { bullets[i].x += bullets[i].vx; bullets[i].y += bullets[i].vy; }
                if (bullets[i].life <= 0) { bullets.splice(i,1); i--; }
            }
        }
        
        function adaptiveSpawn(currentFrame) {
            if (!gameRunning || gameWin) return;
            if (score >= 150 && !finalBossSpawned) { spawnFinalBoss(); return; }
            if (finalBossSpawned) return;
            trySpawnBoss(score);
            let dynamicDelay = Math.max(12, spawnBaseDelay - Math.floor(score / 38));
            if (currentDifficulty === 'easy') dynamicDelay = Math.max(18, dynamicDelay);
            if (currentDifficulty === 'hard') dynamicDelay = Math.max(9, dynamicDelay);
            if (currentFrame - lastSpawnFrame >= dynamicDelay) {
                let spawnCount = 1;
                if (currentDifficulty === 'normal' && score > 20 && Math.random()<0.45) spawnCount=2;
                if (currentDifficulty === 'hard' && score > 15 && Math.random()<0.6) spawnCount=2;
                for(let s=0;s<spawnCount;s++) if (enemies.length + bosses.length < maxEnemiesAtOnce+3) spawnEnemy();
                lastSpawnFrame = currentFrame;
            }
        }
        
        // ==================== DRAW MAIN ====================
        function draw() {
            drawBackground();
            for(let e of enemies) drawEnemy(e, false);
            for(let b of bosses) drawEnemy(b, true);
            
            const px=width/2, py=height/2;
            ctx.shadowBlur=8; ctx.shadowColor="#3cc0ff";
            ctx.beginPath(); ctx.moveTo(px,py-28); ctx.lineTo(px-20,py+10); ctx.lineTo(px-7,py+4);
            ctx.lineTo(px-7,py+20); ctx.lineTo(px,py+14); ctx.lineTo(px+7,py+20); ctx.lineTo(px+7,py+4);
            ctx.lineTo(px+20,py+10); ctx.fillStyle="#3aa8ff"; ctx.fill();
            ctx.beginPath(); ctx.arc(px,py-6,16,0,Math.PI*2); ctx.fillStyle="#1f80ff"; ctx.fill();
            
            ctx.shadowBlur=0;
            ctx.beginPath(); ctx.arc(aimX,aimY,16,0,Math.PI*2); ctx.strokeStyle="#ffec99"; ctx.lineWidth=2.5; ctx.stroke();
            ctx.beginPath(); ctx.moveTo(aimX-26,aimY); ctx.lineTo(aimX-12,aimY); ctx.moveTo(aimX+26,aimY); ctx.lineTo(aimX+12,aimY);
            ctx.moveTo(aimX,aimY-26); ctx.lineTo(aimX,aimY-12); ctx.moveTo(aimX,aimY+26); ctx.lineTo(aimX,aimY+12); ctx.stroke();
            
            for(let b of bullets) {
                ctx.beginPath(); ctx.arc(b.x, b.y, b.radius * (0.6 + (1-b.life)*0.8), 0, Math.PI*2);
                ctx.fillStyle = b.color || `rgba(255, ${80 + (1-b.life)*100}, 20, ${b.life*0.9})`;
                ctx.fill();
            }
            
            if(currentDifficulty && gameRunning && !gameWin){
                ctx.font="bold 14px monospace"; ctx.fillStyle="#facc88";
                ctx.fillText(`MODE: ${currentDifficulty.toUpperCase()}`, width-150, 40);
                if(finalBossSpawned) { ctx.font="bold 18px monospace"; ctx.fillStyle="#ff4444"; ctx.fillText("⚠️ ONE SHOT KILL BOSS ACTIVE! ⚠️", width-290, 75); }
            }
            if(!gameRunning && !gameWin && currentDifficulty){
                ctx.font="bold 48px monospace"; ctx.fillStyle="#ff8866"; ctx.textAlign="center";
                ctx.fillText("💀 GAME OVER 💀", width/2, height/2-60);
                ctx.font="22px monospace"; ctx.fillStyle="#ffddaa";
                ctx.fillText("Tekan 'RESTART MISSION'", width/2, height/2+30);
                ctx.textAlign="left";
            }
            if(gameWin && currentDifficulty){
                ctx.font="bold 46px monospace"; ctx.fillStyle="#ffd966"; ctx.textAlign="center";
                ctx.fillText("🏆 VICTORY! FINAL BOSS SLAIN! 🏆", width/2, height/2-80);
                ctx.font="26px monospace"; ctx.fillStyle="#ffcc88";
                ctx.fillText(`Skor Akhir: ${score}  |  Galaksi Selamat!`, width/2, height/2);
                ctx.font="20px monospace"; ctx.fillStyle="#ffbb66";
                ctx.fillText("Gunakan tombol di bawah untuk 'MAIN MENU' atau 'MAIN LAGI'", width/2, height/2+70);
                ctx.textAlign="left";
            }
            ctx.font="bold 20px monospace"; ctx.fillStyle="#f5e2b0"; ctx.fillText(`👾 ${enemies.length+bosses.length}`, 28, 55);
        }
        
        function gameLoop() {
            if (gameRunning && !gameWin) { updateEntities(); updateBulletEffects(); frameCount++; adaptiveSpawn(frameCount); }
            else { updateBulletEffects(); frameCount++; }
            draw();
            animationId = requestAnimationFrame(gameLoop);
        }
        
        // ==================== MENU & RESET ====================
        function resetCurrentGame() {
            if (!currentDifficulty) return;
            gameRunning = true; gameWin = false; finalBossSpawned = false; score = 0;
            applyDifficultySettings(); enemies = []; bosses = []; bullets = [];
            frameCount = 0; lastSpawnFrame = 0; enemyId = 0; lastBossScoreThreshold = -1;
            scoreSpan.innerText = "0"; updateHealthUI();
            for(let i=0;i<3;i++) spawnEnemy(); lastSpawnFrame = 12;
            if(navigator.vibrate) navigator.vibrate(40); removeOverlay();
        }
        
        function removeOverlay() { const ov = document.querySelector('.menu-overlay'); if(ov) ov.remove(); }

            gameRunning = false; gameWin = false; currentDifficulty = null; finalBossSpawned = false; removeOverlay();

            document.body.appendChild(overlay);
            document.getElementById('easyModeBtn').onclick = () => { currentDifficulty='easy'; applyDifficultySettings(); startGameAfterMode(); };
            document.getElementById('normalModeBtn').onclick = () => { currentDifficulty='normal'; applyDifficultySettings(); startGameAfterMode(); };
            document.getElementById('hardModeBtn').onclick = () => { currentDifficulty='hard'; applyDifficultySettings(); startGameAfterMode(); };
        }
        
        function startGameAfterMode() {
            gameRunning = true; gameWin = false; finalBossSpawned = false; score = 0;
            enemies = []; bosses = []; bullets = []; frameCount = 0; lastSpawnFrame = 0; enemyId = 0; lastBossScoreThreshold = -1;
            scoreSpan.innerText = "0"; updateHealthUI();
            for(let i=0;i<3;i++) spawnEnemy(); lastSpawnFrame = 12;
            infoPanel.style.display = "flex"; controlsTip.style.display = "inline-flex"; removeOverlay();
        }
        
        function showVictoryMenu() {
            if(!gameWin) return;
            const vicOverlay = document.createElement('div'); vicOverlay.className = 'menu-overlay';
            vicOverlay.innerHTML = `<div class="menu-card"><div class="win-message">✨ VICTORY! FINAL BOSS DESTROYED ✨</div><p style="color:#ffdd99;">Skor akhir: ${score} | Mode: ${currentDifficulty.toUpperCase()}</p><div class="action-buttons"><button id="victoryMainMenuBtn" style="background:#2c5a2e;">🏠 MAIN MENU</button><button id="victoryPlayAgainBtn" style="background:#b9772e;">🔄 MAIN LAGI (mode sama)</button></div></div>`;

        }
        
        function attachGameEvents() {
            canvas.addEventListener('mousemove', (e) => { if(gameRunning && !gameWin){ const rect=canvas.getBoundingClientRect(); let cx=(e.clientX-rect.left)*(width/rect.width); let cy=(e.clientY-rect.top)*(height/rect.height); aimX=Math.min(width,Math.max(0,cx)); aimY=Math.min(height,Math.max(0,cy)); } });
            canvas.addEventListener('click', (e) => { if(gameRunning && !gameWin){ const rect=canvas.getBoundingClientRect(); let cx=(e.clientX-rect.left)*(width/rect.width); let cy=(e.clientY-rect.top)*(height/rect.height); shootAt(cx,cy); } });
            canvas.addEventListener('touchmove', (e) => { e.preventDefault(); if(gameRunning && !gameWin){ const rect=canvas.getBoundingClientRect(); let cx=(e.touches[0].clientX-rect.left)*(width/rect.width); let cy=(e.touches[0].clientY-rect.top)*(height/rect.height); aimX=Math.min(width,Math.max(0,cx)); aimY=Math.min(height,Math.max(0,cy)); } });
            canvas.addEventListener('touchstart', (e) => { e.preventDefault(); if(gameRunning && !gameWin){ const rect=canvas.getBoundingClientRect(); let cx=(e.touches[0].clientX-rect.left)*(width/rect.width); let cy=(e.touches[0].clientY-rect.top)*(height/rect.height); shootAt(cx,cy); } });
            resetBtn.addEventListener('click', () => { if(currentDifficulty){ resetCurrentGame(); } else { showMainMenu(); } });
        }
        
        function checkVictoryPopup() { if(gameWin && currentDifficulty && !document.querySelector('.menu-overlay')) showVictoryMenu(); }
        setInterval(checkVictoryPopup, 200);
        
        function init() { initBackground(); attachGameEvents(); gameLoop(); showMainMenu(); }
        init();
    })();
